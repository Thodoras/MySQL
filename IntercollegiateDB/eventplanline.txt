CREATE TABLE EVENTPLANLINE
(	PLANNO VARCHAR(4) NOT NULL,
	LINENO INT NOT NULL,
	TIMESTART DATE NOT NULL,
	TIMEEND DATE NOT NULL,
	NUMBERFLD INT NOT NULL,
	LOCNO VARCHAR(4) NOT NULL,
	RESNO VARCHAR(4) NOT NULL,
	CONSTRAINT PK_EPL PRIMARY KEY (PLANNO, LINENO),
	CONSTRAINT FK_PLANNO_EPL FOREIGN KEY (PLANNO) REFERENCES EVENTPLAN (PLANNO),
	CONSTRAINT FK_LOCNO_EPL FOREIGN KEY (LOCNO) REFERENCES LOCATION (LOCNO),
	CONSTRAINT FK_RESNO_EPL FOREIGN KEY (RESNO) REFERENCES RESOURCETBL (RESNO),
	CONSTRAINT VALID_DATE CHECK (TIMESTART < TIMEEND)
);
