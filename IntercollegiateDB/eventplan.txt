CREATE TABLE EVENTPLAN
(	PLANNO VARCHAR(4) NOT NULL,
	EVENTNO VARCHAR(4) NOT NULL,
	WORKDATE DATE NOT NULL,
	NOTES VARCHAR(50),
	ACTIVITY VARCHAR(21) NOT NULL,
	EMPNO VARCHAR(4),
	CONSTRAINT PK_PLANNO PRIMARY KEY (PLANNO),
	CONSTRAINT FK_EVENTNO_EP FOREIGN KEY (EVENTNO) REFERENCES EVENTREQUEST (EVENTNO),
	CONSTRAINT FK_EMPNO_EP FOREIGN KEY (EMPNO) REFERENCES EMPLOYEE (EMPNO)
);
