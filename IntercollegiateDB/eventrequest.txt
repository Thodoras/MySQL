CREATE TABLE EVENTREQUEST
(	EVENTNO VARCHAR(4) NOT NULL,
	DATEHELD DATE NOT NULL,
	DATEREQ DATE NOT NULL,
	CUSTNO VARCHAR(4) NOT NULL,
	FACNO VARCHAR(4) NOT NULL,
	DATEAUTH DATE,
	STATUS VARCHAR(21) NOT NULL,
	ESTCOST INT NOT NULL,
	ESTAUDIENCE INT NOT NULL,
	BUDNO VARCHAR(5),
	CONSTRAINT PK_EVENTNO PRIMARY KEY (EVENTNO),
	CONSTRAINT FK_CUSTNO_ER FOREIGN KEY (CUSTNO) REFERENCES CUSTOMER (CUSTNO),
	CONSTRAINT FK_FACNO_ER FOREIGN KEY (FACNO) REFERENCES FACILITY (FACNO),
	CONSTRAINT STATUS_DEF CHECK (STATUS IN ("Pending", "Denied", "Approved")),
	CONSTRAINT ESTAUDIENCE_POS CHECK (ESTAUDIENCE > 0)
);
